DB:
# to connect to my database
\c recipe_db;

CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create a dedicated DB user (Do not use the default postgres user for your app)
CREATE USER app_user WITH ENCRYPTED PASSWORD 'SuperSecretPassword123!';
GRANT ALL PRIVILEGES ON DATABASE recipe_db TO app_user;
GRANT ALL PRIVILEGES ON TABLE users TO app_user;
GRANT USAGE, SELECT ON SEQUENCE users_id_seq TO app_user;

sudo find / -name "postgresql.conf"
sudo nano /var/lib/pgsql/data/postgresql.conf
make it = listen_addresses = '*'

sudo nano /var/lib/pgsql/data/pg_hba.conf
new line = host    all             all             10.0.2.0/24 or backend subnet             md5
sudo systemctl restart postgresql

Frontend:
sudo yum install nginx -y
sudo systemctl start nginx
sudo systemctl enable nginx
# Change the ownership of the directory to the ec2-user
sudo chown -R ec2-user:ec2-user /var/www/html

# Ensure the permissions are set correctly
sudo chmod -R 755 /var/www/html

sudo nano /etc/nginx/conf.d/recipe-app.conf
server {
    listen 80;
    server_name _;

    root /var/www/html;
    index home.html index.html index.htm;

    # Serve static frontend files
    location / {
        try_files $uri $uri/ =404;
    }

    # Proxy API requests to your Private Backend EC2
    location /api/ {
        proxy_pass http://10.0.2.x:3000/; # Replace with Backend EC2 Private IP
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }
}
sudo nano /etc/nginx/nginx.conf
then comment out lines that start with server {listen 80; ...}
sudo systemctl restart nginx
